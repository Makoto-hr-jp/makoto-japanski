	\begin{kaitou}[Lv. 2]
		\kai{あたらしい　くつを　かわない。}{Neću kupiti nove cipele.}
		\kai{てがみを　かいた。}{Napisao sam pismo.}
		\kai{ごみを　だす。}{Bacit ću smeće.}
		\kai{ともだちを　またない。}{Neću čekati prijatelja.}
		\kai{とりは　しんだ。}{Ptica je umrla.}
	\end{kaitou}
