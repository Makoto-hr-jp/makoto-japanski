	\dai{Domaća zadaća - Osobne zamjenice i oslovljavanje ljudi; čestice と i や}
	\begin{mondai}{}
		\item Istražite i napišite japanska imena: tri ženska, tri muška i tri koja mogu biti oboje. Nađite neko ime koje se može pisati na dva ili više načina.
		
		    \begin{reibun}
		    	\rei{Ženska:}{あかり、　くみこ（くみ子）、　みう}
			    \rei{Muška:}{だいすけ（太助）、　りゅうた（柳多）、　たつや（起也）}
		    	\rei{Neodređena:}{あき、　りん、　きやう} 
		    	\rei{Imaju više pisanja:}{さくら　さく良　さ久ら　愛咲　叉鞍}
	    	\end{reibun}
    		
		\item Istražite i napišite četri japanska prezimena.　		
\begin{reibun}
 \rei{Prezimena:}{さいじょう（葛城）, もがみ（馬上）, しぶや（渋屋）, まつだ（松田）}
\end{reibun}
		
		\item Nabrojite par komada voća ili povrća koji zajedno idu dobro u salatu koristeći česticu と.
		
		\begin{reibun}
			\rei{Nabrajanje sa と:}{りんご 　と 　ぶどう 　と 　いちご 　と 　あんず}
		\end{reibun}
		
		\item Nabrojite par stvari koje ne idu dobro u salatu koristeći česticu や.
		
		\begin{reibun}
			\rei{Nabrajanje sa や:}{しお　や　 マヨネーズ　や　にんじん　や　きゅうり}
		\end{reibun}
		
	\end{mondai}
	
	\begin{mondai}{Pročitajte ova imena i prezimena te odgonetnite je li ta osoba mlađa, starija ili nije definirano.}
		\item はしもと　けんじさん - nije definirano
		\item おぎわら　ももこちゃん - mlađa
		\item とうま　まさみくん - mlađa
		\item えぐちせんぱい - starija
		\item なかむらしゃちょう - starija
	\end{mondai}
