	\end{kaitou}

    \begin{kaitou}[Lv. 2]
    	\kainagai{あの木の　上に　ねこが　いる。}{Ispod onog drveta je mačka.}
    	\kainagai{この　きっさてんは　がっこうから　とおい。}{Ovaj kafić je daleko od škole.}
    	\kainagai{テーブルの下の　はこの中は　くらかった。}{U kutiji ispod stola je mračno.}
    	\kainagai{たけしくんは　ここから　がっこうまで　行った。}{Takeši je otišao odavdje do škole.}
    	\kainagai{川の　むこうから　ぎんこうの　となりの　みせまで　来た。}{S druge strane rijeke došla sam do kafića pored banke.}
    \end{kaitou}

    \begin{kaitou}[Lv. 3]
    	\kainagai{あの　まつの　木の　下に　くろい　ねこが　いた。}{Ispod onog bora je crna mačka.}

    	\kainagai{テーブルの下の　はこの中に たけしくんの　ともだちの　こいぬが　いた。}{U kutiji ispod stola bio je psić Takešijevog prijatelja.}
    	\kainagai{たけしくんは　ゆうびんきょくの　となりの　みせから　がっこうまで　行った。}{Takeši je od dućana pored pošte otišao do škole.}
    	\kainagai{川の　むこうから　ぎんこうの　となりの　みせの　みぎの　きっさてんまで　来た。}{Sa suprotne strane rijeke došla sam do kafića s desne strane dućana koji je pored banke.}
    \end{kaitou}
	
	\begin{kaitou}[Lv. 4*]

		\kainagai{川の　むこうから　ぎんこうの　となりに　ある　みせの　みぎに　ある　きっさてんまで　来た。}{S druge strane rijeke došla sam do kafića s desne strane dućana u blizini banke.}
