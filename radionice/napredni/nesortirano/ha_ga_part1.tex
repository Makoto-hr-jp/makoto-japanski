		--->	
		\rei{B:これ\underline{は}私\underline{が}東京で買った時計です。}{Ovo je sat kojeg sam kupio u Tokiju.}
	\end{reibun}

		--->
		\rei{B:ひらがな\underline{は}書けますが、漢字\underline{は}書けません}{Hiraganu znam pisati, ali kanji ne znam.}
	\end{reibun}
